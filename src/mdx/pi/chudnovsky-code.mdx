  const D = 10005n
  const Y_MIN = 10n ** BigInt(digits / 2 + 5)
  let [x1, y1] = [1n, 0n]
  let [x2, y2] = [4001n, 40n]
  while (true) {
    const [x, y] = [x1 * x2 + D * y1 * y2, x1 * y2 + y1 * x2]
    ;[x1, y1, x2, y2] = [x2, y2, x, y]
    if (y > Y_MIN) break
  }
  return [x2, y2]
}
